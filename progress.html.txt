  <script type="module">
    import { db } from "./firebase.js";
    import { doc, onSnapshot } from
      "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";

    const TOTAL = 100;

    function update(house, value) {
      const v = Math.min(value || 0, TOTAL);
      document.getElementById(house + "-bar").style.width = v + "%";
      document.getElementById(house + "-text").textContent = v + "%";
    }

    const ref = doc(db, "scores", "houses");

    onSnapshot(ref, snap => {
      if (!snap.exists()) return;
      const d = snap.data();

      update("gryffindor", d.gryffindor);
      update("slytherin",  d.slytherin);
      update("ravenclaw",  d.ravenclaw);
      update("hufflepuff", d.hufflepuff);
    });
  </script>
